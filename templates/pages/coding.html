    display(array, size);    
    int i, j, temp; 
    for (i = 0; i < size-1; i++){ 
    for (j = 0; j < size-i-1; j++) if (array[j] > array[j+1]) 
    {
    temp = array[j]; // swap the element
    array[j] = array[j+1]; 
    array[j+1] = temp;

    }

C program to print the full pyramid pattern of alphabets

  #include 
  int main()
  {
    int rows;
    printf("Number of rows: ");
    scanf("%d", &rows);

    for (int i = 1; i <= rows; i++) {
  
      // inner loop 1 to print white spaces
      for (int j = 1; j <= 2 * (rows - i); j++) {
        printf(" ");
      }
  
      for (int k = 1; k < 2 * i; k++) {
        printf("%d ", i);
      }
      printf("\n");
    }
    return 0;
  }
